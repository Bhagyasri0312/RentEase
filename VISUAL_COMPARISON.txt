COLOR SCHEME - Design Details
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

IMAGE SECTION:
  Background: Linear gradient (to-br from-blue-100 to-blue-50)
              Light blue #F0F9FF → Even lighter #EFF6FF
  Overlay:    from-black/40 via-black/20 to-transparent
              Progressive darkening for text readability
              Normal opacity: 50%
              Hover opacity: 60% (darker for emphasis)

CONTENT SECTION:
  Background: White #FFFFFF
  Title text: Slate-900 (dark gray #0F172A)
  Count text: Slate-600 (medium gray #475569)
  
  ON HOVER:
    Title color: Blue-600 (brand blue #2563EB)
    Action text: Blue-600 with arrow icon
    Action opacity: 0% → 100% fade-in

SHADOW & ELEVATION:
  Normal: shadow-lg
          0 10px 15px -3px rgba(0, 0, 0, 0.1)
  Hover:  shadow-2xl
          0 25px 50px -12px rgba(0, 0, 0, 0.25)
